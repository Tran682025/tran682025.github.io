import express from "express"; import fetch from "node-fetch"; import dotenv from "dotenv"; dotenv.config(); const app = express(); app.use(express.json()); const API = "https://api.minepi.com/v2"; const { PI_API_KEY, APP_ID, PORT = 3000 } = process.env; async function piFetch(path, options = {}) { const res = await fetch(`${API}${path}`, { ...options, headers: { "Content-Type": "application/json", Authorization: `Key ${PI_API_KEY}`, ...(options.headers || {}) } }); if (!res.ok) throw new Error(`${res.status} ${await res.text()}`); return res.json(); } // 1) Approve app.post("/approve", async (req, res) => { try { const { paymentId } = req.body; // (optional) verify payment belongs to your APP_ID const payment = await piFetch(`/payments/${paymentId}`, { method: "GET" }); if (payment.app_id !== APP_ID) throw new Error("Wrong APP_ID"); const approved = await piFetch(`/payments/${paymentId}/approve`, { method: "POST" }); res.json({ ok: true, approved }); } catch (e) { console.error(e); res.status(500).json({ ok: false, error: e.message }); } }); // 2) Complete app.post("/complete", async (req, res) => { try { const { paymentId, txid } = req.body; const completed = await piFetch(`/payments/${paymentId}/complete`, { method: "POST", body: JSON.stringify({ txid }) }); res.json({ ok: true, completed }); } catch (e) { console.error(e); res.status(500).json({ ok: false, error: e.message }); } }); app.get("/", (req, res) => res.send("Pi backend up")); app.listen(PORT, () => console.log("Server running on port", PORT));
